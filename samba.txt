==安裝samba============================================

sudo apt-get install samba -y

****更改設定
sudo vi /etc/samba/smb.conf

//更改
workgroup = XXX
map to guest = Never

//加入檔案尾端
[Public]
path = /home/UserName/Public
valid users = <USER>
available = yes
browseable = yes
read only = no
writable = yes
public = yes
create mask = 0660
directory mask = 2770

****重啟samba
sudo service smbd restart

****重新設定已存在的檔案權限
sudo find <SHARE_DIR_PATH> -type d -print0 | xargs -0 chmod 2770   # change dirs
sudo find <SHARE_DIR_PATH> -type f -print0 | xargs -0 chmod 660    # change files
sudo chown <USER>:<USERGROUP> <FILEPATH>                     # change owner

****check group of account
grouops <USER>

****windows 清除連線帳密
net use * /delete
