==安裝samba============================================

sudo apt-get install samba -y

****更改設定
sudo vi /etc/samba/smb.conf

//更改
workgroup = XXX
map to guest = Never

//加入檔案尾端
[Public]
path = /home/UserName/Public
write list = XXX
valid users = XXX
available = yes
browseable = yes
read only = no
writable = yes
public = yes
create mask = 0644          #0660
directory mask = 0755       #2770

****重啟samba
sudo service smbd restart
